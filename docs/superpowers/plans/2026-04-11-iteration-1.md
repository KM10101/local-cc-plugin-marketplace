```typescript
import { Worker } from 'worker_threads'
import { join, dirname } from 'path'
import { fileURLToPath } from 'url'
import { existsSync } from 'fs'
import type { Db } from '../db.js'
import type { Task } from '../types.js'

const __dirname = dirname(fileURLToPath(import.meta.url))

function now() { return new Date().toISOString() }

export interface SchedulerOptions {
  maxConcurrent?: number
}

export class TaskScheduler {
  private db: Db
  private maxConcurrent: number
  private workers = new Map<string, Worker>()

  constructor(db: Db, options: SchedulerOptions = {}) {
    this.db = db
    this.maxConcurrent = options.maxConcurrent ?? 20
  }

  getRunningCount(): number {
    const row = this.db.prepare(`SELECT COUNT(*) as c FROM tasks WHERE status = 'running'`).get() as any
    return row.c
  }

  getNextQueued(): Task | null {
    return (this.db.prepare(`SELECT * FROM tasks WHERE status = 'queued' ORDER BY created_at ASC LIMIT 1`).get() as Task) ?? null
  }

  hasActiveTask(repoUrl: string, branch: string): boolean {
    const row = this.db.prepare(
      `SELECT COUNT(*) as c FROM tasks WHERE repo_url = ? AND branch = ? AND status IN ('running', 'queued')`
    ).get(repoUrl, branch) as any
    return row.c > 0
  }

  /**
   * Try to start queued tasks up to the concurrency limit.
   * Called after enqueue, after a task completes, and after resume.
   */
  drainQueue(): void {
    while (this.getRunningCount() < this.maxConcurrent) {
      const next = this.getNextQueued()
      if (!next) break
      this.startWorker(next)
    }
  }

  /**
   * Enqueue a task (it's already inserted as 'queued' in DB) and try to drain.
   */
  enqueue(taskId: string): void {
    this.drainQueue()
  }

  /**
   * Start a worker for a task. Updates status to 'running'.
   */
  private startWorker(task: Task): void {
    this.db.prepare(`UPDATE tasks SET status = 'running' WHERE id = ?`).run(task.id)

    const isPlugin = task.type === 'clone_plugin'
    const workerJsPath = join(__dirname, '..', 'workers', 'clone-worker.js')
    const workerTsPath = workerJsPath.replace(/\.js$/, '.ts')
    const useTs = !existsSync(workerJsPath) && existsSync(workerTsPath)
    const workerPath = useTs ? workerTsPath : workerJsPath

    const workerData = isPlugin
      ? { mode: 'plugin', taskId: task.id, pluginId: task.plugin_id, sourceUrl: task.repo_url, branch: task.branch }
      : { mode: 'marketplace', taskId: task.id, marketplaceId: task.marketplace_id, sourceUrl: task.repo_url, branch: task.branch }

    const worker = new Worker(workerPath, {
      workerData,
      execArgv: useTs ? ['--import', 'tsx/esm'] : [],
    })

    this.workers.set(task.id, worker)

    worker.on('message', (msg: any) => {
      this.handleWorkerMessage(task, msg)
    })

    worker.on('error', (err) => {
      this.db.prepare(`UPDATE tasks SET status = 'failed', message = ?, completed_at = ? WHERE id = ?`)
        .run(err.message, now(), task.id)
      this.workers.delete(task.id)
      this.updateParentStatus(task.parent_task_id)
      this.drainQueue()
    })

    worker.on('exit', (code) => {
      this.workers.delete(task.id)
      if (code !== 0) {
        const current = this.db.prepare(`SELECT status FROM tasks WHERE id = ?`).get(task.id) as any
        if (current && current.status === 'running') {
          this.db.prepare(`UPDATE tasks SET status = 'failed', message = ?, completed_at = ? WHERE id = ?`)
            .run(`Worker exited with code ${code}`, now(), task.id)
        }
        this.updateParentStatus(task.parent_task_id)
      }
      this.drainQueue()
    })
  }

  private handleWorkerMessage(task: Task, msg: any): void {
    if (msg.type === 'progress') {
      this.db.prepare(`UPDATE tasks SET progress = ?, message = ? WHERE id = ?`)
        .run(msg.progress, msg.message, task.id)
      if (task.parent_task_id) {
        this.updateParentProgress(task.parent_task_id)
      }
    } else if (msg.type === 'done') {
      this.db.prepare(`UPDATE tasks SET status = 'completed', progress = 100, message = ?, completed_at = ? WHERE id = ?`)
        .run(msg.message ?? 'Completed', now(), task.id)
      this.workers.delete(task.id)
      if (msg.onDone && typeof msg.onDone === 'function') {
        msg.onDone()
      }
      this.updateParentStatus(task.parent_task_id)
      this.drainQueue()
    } else if (msg.type === 'error') {
      this.db.prepare(`UPDATE tasks SET status = 'failed', message = ?, completed_at = ? WHERE id = ?`)
        .run(msg.message, now(), task.id)
      this.workers.delete(task.id)
      this.updateParentStatus(task.parent_task_id)
      this.drainQueue()
    } else if (msg.type === 'create_child_tasks') {
      // Worker requests creation of child tasks for external plugins
      for (const child of msg.children as any[]) {
        const childId = crypto.randomUUID()
        this.db.prepare(`INSERT INTO tasks (id, parent_task_id, type, status, marketplace_id, repo_url, branch, plugin_id, progress, created_at)
          VALUES (?, ?, 'clone_plugin', 'queued', ?, ?, ?, ?, 0, ?)`
        ).run(childId, task.id, task.marketplace_id, child.sourceUrl, task.branch, child.pluginId, now())
      }
      this.drainQueue()
    } else if (msg.type === 'marketplace_done') {
      // Marketplace repo itself is cloned; persist results, then let child tasks handle plugins
      // This is called by the worker after marketplace clone + parse
      // msg contains: { gitSha, plugins, marketplaceId }
      // We store this so persistCloneResults can be called
      this.handleMarketplaceDone(task, msg)
    }
  }

  private handleMarketplaceDone(task: Task, msg: any): void {
    // This will be called from marketplace-service via a callback pattern
    // For now, emit an event that marketplace-service listens to
    if (this.onMarketplaceDone) {
      this.onMarketplaceDone(task, msg)
    }
  }

  /**
   * Callback set by marketplace-service to handle clone completion.
   */
  onMarketplaceDone: ((task: Task, msg: any) => void) | null = null

  private updateParentProgress(parentTaskId: string | null): void {
    if (!parentTaskId) return
    const children = this.db.prepare(`SELECT progress FROM tasks WHERE parent_task_id = ?`).all(parentTaskId) as any[]
    if (children.length === 0) return
    // Parent's own marketplace clone counts as 1 unit; each child is 1 unit
    // We approximate: parent progress = average of children progress
    // The parent's own clone progress is tracked in the parent task itself when it was running
    const parentRow = this.db.prepare(`SELECT progress FROM tasks WHERE id = ?`).get(parentTaskId) as any
    if (!parentRow) return
    const total = children.reduce((sum: number, c: any) => sum + c.progress, 0)
    const avgProgress = Math.round(total / children.length)
    // Parent progress: blend own marketplace clone (assume done = 100 weighted as 1) with children
    const blended = Math.round((100 + total) / (1 + children.length))
    this.db.prepare(`UPDATE tasks SET progress = ? WHERE id = ?`).run(blended, parentTaskId)
  }

  updateParentStatus(parentTaskId: string | null): void {
    if (!parentTaskId) return
    const children = this.db.prepare(`SELECT status FROM tasks WHERE parent_task_id = ?`).all(parentTaskId) as any[]
    if (children.length === 0) return

    const statuses = children.map((c: any) => c.status)
    let newStatus: string

    if (statuses.some((s: string) => s === 'running')) {
      newStatus = 'running'
    } else if (statuses.every((s: string) => s === 'completed')) {
      newStatus = 'completed'
      this.db.prepare(`UPDATE tasks SET status = ?, progress = 100, completed_at = ? WHERE id = ?`)
        .run(newStatus, now(), parentTaskId)
      return
    } else if (statuses.every((s: string) => s === 'stopped' || s === 'queued')) {
      newStatus = 'stopped'
    } else if (statuses.some((s: string) => s === 'failed') && !statuses.some((s: string) => s === 'running')) {
      newStatus = 'failed'
    } else {
      newStatus = 'running'
    }

    this.db.prepare(`UPDATE tasks SET status = ? WHERE id = ?`).run(newStatus, parentTaskId)
  }

  stopTask(taskId: string): void {
    const task = this.db.prepare(`SELECT * FROM tasks WHERE id = ?`).get(taskId) as Task | undefined
    if (!task) return

    // Stop the worker if running
    const worker = this.workers.get(taskId)
    if (worker) {
      worker.terminate()
      this.workers.delete(taskId)
    }

    this.db.prepare(`UPDATE tasks SET status = 'stopped' WHERE id = ?`).run(taskId)

    // If parent, stop all running/queued children
    const children = this.db.prepare(`SELECT id FROM tasks WHERE parent_task_id = ? AND status IN ('running', 'queued')`).all(taskId) as any[]
    for (const child of children) {
      const childWorker = this.workers.get(child.id)
      if (childWorker) {
        childWorker.terminate()
        this.workers.delete(child.id)
      }
      this.db.prepare(`UPDATE tasks SET status = 'stopped' WHERE id = ?`).run(child.id)
    }
  }

  resumeTask(taskId: string): void {
    const task = this.db.prepare(`SELECT * FROM tasks WHERE id = ?`).get(taskId) as Task | undefined
    if (!task || (task.status !== 'stopped' && task.status !== 'failed')) return

    this.db.prepare(`UPDATE tasks SET status = 'queued' WHERE id = ?`).run(taskId)

    // If parent, resume all stopped/failed children
    const children = this.db.prepare(`SELECT id FROM tasks WHERE parent_task_id = ? AND status IN ('stopped', 'failed')`).all(taskId) as any[]
    for (const child of children) {
      this.db.prepare(`UPDATE tasks SET status = 'queued' WHERE id = ?`).run(child.id)
    }

    this.drainQueue()
  }

  deleteTask(taskId: string): void {
    // Stop workers first
    const worker = this.workers.get(taskId)
    if (worker) {
      worker.terminate()
      this.workers.delete(taskId)
    }

    const children = this.db.prepare(`SELECT id FROM tasks WHERE parent_task_id = ?`).all(taskId) as any[]
    for (const child of children) {
      const childWorker = this.workers.get(child.id)
      if (childWorker) {
        childWorker.terminate()
        this.workers.delete(child.id)
      }
    }

    // Delete children first, then parent
    this.db.prepare(`DELETE FROM tasks WHERE parent_task_id = ?`).run(taskId)
    this.db.prepare(`DELETE FROM tasks WHERE id = ?`).run(taskId)
  }

  shutdown(): void {
    for (const [id, worker] of this.workers) {
      worker.terminate()
    }
    this.workers.clear()
  }
}
```

```typescript
import { join } from 'path'
import { rm } from 'fs/promises'
import { existsSync } from 'fs'
import type { Db } from '../db.js'
import { readPluginJson, parseMarketplaceJson } from './plugin-service.js'
import type { TaskScheduler } from './task-scheduler.js'
import type { ClonePluginResult } from '../workers/clone-worker.js'

function now() { return new Date().toISOString() }
function uuid() { return crypto.randomUUID() }

function repoNameFromUrl(url: string): string {
  const parts = url.replace(/\.git$/, '').split('/')
  return parts[parts.length - 1] || url
}

export interface AddMarketplaceOptions {
  repoUrl: string
  branch?: string
  reposDir: string
}

export function addMarketplace(
  db: Db,
  scheduler: TaskScheduler,
  options: AddMarketplaceOptions,
): { marketplace_id: string; task_id: string } {
  const { repoUrl, branch = 'main', reposDir } = options

  // Validate: no duplicate repo_url + branch
  const existing = db.prepare(`SELECT id FROM marketplaces WHERE repo_url = ? AND branch = ?`).get(repoUrl, branch)
  if (existing) {
    throw new ConflictError('Marketplace with this URL and branch already exists')
  }

  // Validate: max 5 branches per repo_url
  const branchCount = (db.prepare(`SELECT COUNT(*) as c FROM marketplaces WHERE repo_url = ?`).get(repoUrl) as any).c
  if (branchCount >= 5) {
    throw new ConflictError('Maximum 5 branches per repository')
  }

  // Validate: no active task for this repo_url + branch
  if (scheduler.hasActiveTask(repoUrl, branch)) {
    throw new ConflictError('A clone task for this repository and branch is already in progress')
  }

  const marketplace_id = uuid()
  const task_id = uuid()
  const localPath = join(reposDir, 'marketplaces', marketplace_id)

  db.prepare(`INSERT INTO marketplaces (id, repo_url, branch, name, local_path, status, created_at)
    VALUES (?, ?, ?, ?, ?, 'pending', ?)`
  ).run(marketplace_id, repoUrl, branch, repoNameFromUrl(repoUrl), localPath, now())

  db.prepare(`UPDATE marketplaces SET status='cloning' WHERE id=?`).run(marketplace_id)

  db.prepare(`INSERT INTO tasks (id, parent_task_id, type, status, marketplace_id, repo_url, branch, progress, created_at)
    VALUES (?, null, 'clone_marketplace', 'queued', ?, ?, ?, 0, ?)`
  ).run(task_id, marketplace_id, repoUrl, branch, now())

  scheduler.enqueue(task_id)

  return { marketplace_id, task_id }
}

export function refreshMarketplace(
  db: Db,
  scheduler: TaskScheduler,
  id: string,
  reposDir: string,
): { marketplace_id: string; task_id: string } | null {
  const marketplace = db.prepare(`SELECT * FROM marketplaces WHERE id=?`).get(id) as any
  if (!marketplace) return null

  if (scheduler.hasActiveTask(marketplace.repo_url, marketplace.branch)) {
    throw new ConflictError('A clone task for this repository and branch is already in progress')
  }

  const task_id = uuid()

  db.prepare(`UPDATE marketplaces SET status='cloning' WHERE id=?`).run(id)

  db.prepare(`INSERT INTO tasks (id, parent_task_id, type, status, marketplace_id, repo_url, branch, progress, created_at)
    VALUES (?, null, 'clone_marketplace', 'queued', ?, ?, ?, 0, ?)`
  ).run(task_id, id, marketplace.repo_url, marketplace.branch, now())

  scheduler.enqueue(task_id)

  return { marketplace_id: id, task_id }
}

export async function persistCloneResults(db: Db, marketplace_id: string, gitSha: string, plugins: ClonePluginResult[]) {
  const marketplace = db.prepare(`SELECT local_path FROM marketplaces WHERE id=?`).get(marketplace_id) as { local_path: string }

  let name = marketplace_id
  let description: string | null = null
  let owner: string | null = null
  try {
    const meta = await parseMarketplaceJson(marketplace.local_path)
    name = meta.name || name
    description = meta.description
    owner = meta.owner
  } catch { /* use defaults */ }

  db.prepare(`UPDATE marketplaces SET name=?, description=?, owner=?, status='ready', git_commit_sha=?, last_updated=? WHERE id=?`)
    .run(name, description, owner, gitSha, now(), marketplace_id)

  // Delete existing plugins (re-sync)
  db.prepare(`DELETE FROM plugins WHERE marketplace_id=?`).run(marketplace_id)

  for (const p of plugins) {
    let pluginMeta = { version: null, author: null, author_url: null, description: null, keywords: null, homepage: null, license: null } as any
    try {
      pluginMeta = await readPluginJson(p.local_path)
    } catch { /* use defaults */ }

    const status = p.git_commit_sha !== null || p.source_type === 'local' ? 'ready' : 'error'
    db.prepare(`INSERT INTO plugins (id, marketplace_id, name, version, author, author_url, description, keywords, homepage, license, source_type, source_url, local_path, status, git_commit_sha, created_at)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`
    ).run(
      uuid(), marketplace_id, p.name,
      pluginMeta.version, pluginMeta.author, pluginMeta.author_url,
      pluginMeta.description, pluginMeta.keywords, pluginMeta.homepage, pluginMeta.license,
      p.source_type, p.source_url, p.local_path, status, p.git_commit_sha, now()
    )
  }
}

export function deleteMarketplace(db: Db, id: string, reposDir: string): boolean {
  const row = db.prepare(`SELECT local_path FROM marketplaces WHERE id=?`).get(id) as { local_path: string } | undefined
  if (!row) return false

  // Delete associated plugins manually (no FK)
  db.prepare(`DELETE FROM plugins WHERE marketplace_id=?`).run(id)
  db.prepare(`DELETE FROM marketplaces WHERE id=?`).run(id)

  const pluginDir = join(reposDir, 'plugins', id)
  if (existsSync(row.local_path)) rm(row.local_path, { recursive: true, force: true }).catch(() => {})
  if (existsSync(pluginDir)) rm(pluginDir, { recursive: true, force: true }).catch(() => {})
  return true
}

export function listMarketplaces(db: Db, search?: string) {
  let sql = `
    SELECT m.*, COUNT(p.id) as plugin_count
    FROM marketplaces m
    LEFT JOIN plugins p ON p.marketplace_id = m.id
  `
  const params: any[] = []
  if (search) {
    sql += ` WHERE m.name LIKE ?`
    params.push(`%${search}%`)
  }
  sql += ` GROUP BY m.id ORDER BY m.created_at DESC`
  return db.prepare(sql).all(...params)
}

export function getMarketplace(db: Db, id: string) {
  const marketplace = db.prepare('SELECT * FROM marketplaces WHERE id = ?').get(id) as any
  if (!marketplace) return null

  // Find siblings (same repo_url, different branch)
  const siblings = db.prepare(
    `SELECT id, branch, status FROM marketplaces WHERE repo_url = ? AND id != ? ORDER BY branch`
  ).all(marketplace.repo_url, id)

  return { ...marketplace, siblings }
}

export function getMarketplacePlugins(db: Db, marketplace_id: string, search?: string) {
  let sql = `SELECT * FROM plugins WHERE marketplace_id = ?`
  const params: any[] = [marketplace_id]
  if (search) {
    sql += ` AND name LIKE ?`
    params.push(`%${search}%`)
  }
  sql += ` ORDER BY name`
  return db.prepare(sql).all(...params)
}

export function getRepoBranches(db: Db, repoUrl: string) {
  return db.prepare(`SELECT id, branch, status, name FROM marketplaces WHERE repo_url = ? ORDER BY branch`).all(repoUrl)
}

export class ConflictError extends Error {
  constructor(message: string) {
    super(message)
    this.name = 'ConflictError'
  }
}
```

```typescript
import { Router } from 'express'
import type { Db } from '../db.js'
import type { Task, TaskStatus } from '../types.js'
import type { TaskScheduler } from '../services/task-scheduler.js'

function isTerminal(status: TaskStatus): boolean {
  return status === 'completed' || status === 'failed'
}

export function tasksRouter(db: Db, scheduler: TaskScheduler) {
  const router = Router()

  router.get('/', (req, res) => {
    const search = req.query.search as string | undefined

    let sql = `SELECT * FROM tasks WHERE parent_task_id IS NULL`
    const params: any[] = []
    if (search) {
      sql += ` AND repo_url LIKE ?`
      params.push(`%${search}%`)
    }
    sql += ` ORDER BY created_at DESC`

    const parents = db.prepare(sql).all(...params) as Task[]

    // Attach children
    const withChildren = parents.map(parent => {
      const children = db.prepare(
        `SELECT * FROM tasks WHERE parent_task_id = ? ORDER BY created_at ASC`
      ).all(parent.id) as Task[]
      return { ...parent, children }
    })

    res.json(withChildren)
  })

  router.post('/:id/stop', (req, res) => {
    const task = db.prepare(`SELECT * FROM tasks WHERE id = ?`).get(req.params.id) as Task | undefined
    if (!task) return res.status(404).json({ error: 'Task not found' })
    scheduler.stopTask(req.params.id)
    res.json({ ok: true })
  })

  router.post('/:id/resume', (req, res) => {
    const task = db.prepare(`SELECT * FROM tasks WHERE id = ?`).get(req.params.id) as Task | undefined
    if (!task) return res.status(404).json({ error: 'Task not found' })
    scheduler.resumeTask(req.params.id)
    res.json({ ok: true })
  })

  router.delete('/:id', (req, res) => {
    const task = db.prepare(`SELECT * FROM tasks WHERE id = ?`).get(req.params.id) as Task | undefined
    if (!task) return res.status(404).json({ error: 'Task not found' })
    scheduler.deleteTask(req.params.id)
    res.status(204).send()
  })

  router.get('/:id/events', (req, res) => {
    const task = db.prepare(`SELECT * FROM tasks WHERE id=?`).get(req.params.id) as Task | undefined
    if (!task) return res.status(404).json({ error: 'Task not found' })

    res.setHeader('Content-Type', 'text/event-stream')
    res.setHeader('Cache-Control', 'no-cache')
    res.setHeader('Connection', 'keep-alive')
    res.flushHeaders()

    const send = (data: object) => res.write(`data: ${JSON.stringify(data)}\n\n`)

    send(task)

    if (isTerminal(task.status)) {
      res.end()
      return
    }

    const interval = setInterval(() => {
      const updated = db.prepare(`SELECT * FROM tasks WHERE id=?`).get(req.params.id) as Task | undefined
      if (!updated) { clearInterval(interval); res.end(); return }

      // Also fetch children for parent tasks
      if (!updated.parent_task_id) {
        const children = db.prepare(`SELECT * FROM tasks WHERE parent_task_id = ? ORDER BY created_at ASC`).all(updated.id) as Task[]
        send({ ...updated, children })
      } else {
        send(updated)
      }

      if (isTerminal(updated.status)) {
        clearInterval(interval)
        res.end()
      }
    }, 500)

    req.on('close', () => clearInterval(interval))
  })

  return router
}
```

```typescript
import { workerData, parentPort } from 'worker_threads'

import { join } from 'path'
import { existsSync } from 'fs'
import { mkdir, rm } from 'fs/promises'
import { parseMarketplaceJson } from '../services/plugin-service.js'

export interface CloneWorkerInput {
  mode: 'marketplace' | 'plugin'
  taskId: string
  marketplaceId?: string
  pluginId?: string
  sourceUrl: string
  branch?: string
  reposDir?: string
  pluginDir?: string
}

export type CloneWorkerMessage =
  | { type: 'progress'; progress: number; message: string }
  | { type: 'done'; gitSha: string; plugins: ClonePluginResult[] }
  | { type: 'error'; message: string }

export interface ClonePluginResult {
  name: string
  source_type: 'local' | 'external'
  source_url: string | null
  local_path: string
  relative_path: string
  git_commit_sha: string | null
}

function post(msg: CloneWorkerMessage) {
  parentPort?.postMessage(msg)
}

async function getHeadSha(repoPath: string): Promise<string | null> {
  try {
    const git = simpleGit(repoPath)
    const log = await git.log({ maxCount: 1 })
    return log.latest?.hash ?? null
  } catch {
    return null
  }
}

async function isValidGitRepo(dir: string): Promise<boolean> {
  if (!existsSync(join(dir, '.git'))) return false
  try {
    const git = simpleGit(dir)
    await git.status()
    return true
  } catch {
    return false
  }
}

async function cloneOrPull(sourceUrl: string, targetDir: string, branch?: string): Promise<void> {
  if (await isValidGitRepo(targetDir)) {
    // Existing valid repo — fetch and checkout branch
    const git = simpleGit(targetDir)
    await git.fetch('origin')
    if (branch) {
      await git.checkout(branch).catch(() => git.checkoutBranch(branch, `origin/${branch}`))
    }
    await git.pull('origin', branch || 'HEAD')
  } else {
    // Not a valid repo — clean up and clone fresh
    if (existsSync(targetDir)) {
      await rm(targetDir, { recursive: true, force: true })
    }
    const args = branch ? ['--branch', branch] : []
    await simpleGit().clone(sourceUrl, targetDir, args)
  }
}

async function runMarketplace() {
  const { marketplaceId, sourceUrl, branch, reposDir } = workerData as any

  if (!marketplaceId || !sourceUrl || !reposDir) {
    post({ type: 'error', message: 'Invalid workerData: marketplaceId, sourceUrl, and reposDir are required' })
    return
  }

  const marketplaceDir = join(reposDir, 'marketplaces', marketplaceId)

  try {
    post({ type: 'progress', progress: 5, message: `Cloning marketplace from ${sourceUrl} (branch: ${branch || 'default'})` })
    await mkdir(join(reposDir, 'marketplaces'), { recursive: true })
    await cloneOrPull(sourceUrl, marketplaceDir, branch)

    post({ type: 'progress', progress: 40, message: 'Marketplace cloned, reading plugin list' })

    const marketplaceMeta = await parseMarketplaceJson(marketplaceDir)
    const externalPlugins = marketplaceMeta.plugins.filter(p => p.source_type === 'external')
    const localPlugins = marketplaceMeta.plugins.filter(p => p.source_type === 'local')

    const results: ClonePluginResult[] = []

    for (const p of localPlugins) {
      const localPath = join(marketplaceDir, p.relative_path)
      results.push({
        name: p.name,
        source_type: 'local',
        source_url: null,
        local_path: localPath,
        relative_path: p.relative_path,
        git_commit_sha: null,
      })
    }

    // Clone external plugins
    const pluginsBaseDir = join(reposDir, 'plugins', marketplaceId)
    await mkdir(pluginsBaseDir, { recursive: true })

    for (let i = 0; i < externalPlugins.length; i++) {
      const plugin = externalPlugins[i]
      const pluginDir = join(pluginsBaseDir, plugin.name)
      const progressPct = 40 + Math.round(((i + 1) / externalPlugins.length) * 55)

      post({
        type: 'progress',
        progress: progressPct,
        message: `Cloning plugin ${i + 1}/${externalPlugins.length}: ${plugin.name}`,
      })

      if (!plugin.source_url) {
        results.push({
          name: plugin.name,
          source_type: 'external',
          source_url: null,
          local_path: pluginDir,
          relative_path: plugin.relative_path,
          git_commit_sha: null,
        })
        continue
      }

      try {
        await cloneOrPull(plugin.source_url, pluginDir)
        const sha = await getHeadSha(pluginDir)
        results.push({
          name: plugin.name,
          source_type: 'external',
          source_url: plugin.source_url,
          local_path: pluginDir,
          relative_path: plugin.relative_path,
          git_commit_sha: sha,
        })
      } catch (err: any) {
        post({
          type: 'progress',
          progress: progressPct,
          message: `Warning: failed to clone ${plugin.name}: ${err.message}`,
        })
        results.push({
          name: plugin.name,
          source_type: 'external',
          source_url: plugin.source_url,
          local_path: pluginDir,
          relative_path: plugin.relative_path,
          git_commit_sha: null,
        })
      }
    }

    const marketplaceGitSha = await getHeadSha(marketplaceDir)
    post({ type: 'done', gitSha: marketplaceGitSha ?? '', plugins: results })
  } catch (err: any) {
    post({ type: 'error', message: err.message ?? String(err) })
  }
}
```

```tsx
let nextId = 0
```
